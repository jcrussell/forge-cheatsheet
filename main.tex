% --- Section: Core Nav ---
\section{Focus \& Navigation}
\textit{Navigate the window tree using Vim-style keys.}

\begin{tabular}{@{}ll@{}}
    \textbf{Focus Window} & \\
    \cmd + \key{h} \key{j} \key{k} \key{l} & Left / Down / Up / Right \\
    \textbf{Move Window} & \textit{(Moves across monitors)} \\
    \cmd + \shft + \key{h} \key{j} \key{k} \key{l} & Move in direction \\
    \textbf{Swap Window} & \textit{(In-place exchange)} \\
    \cmd + \ctrl + \key{h} \key{j} \key{k} \key{l} & Swap with neighbor \\
    \cmd + \key{Enter} & Swap with Master \\
\end{tabular}

\vspace{4pt}

% --- Section: Window Lifecycle ---
\section{Window Lifecycle}
\textit{Managing windows and applications.}

\begin{tabular}{@{}ll@{}}
    \textbf{Close \& Quit} & \\
    \alt + \key{F4} & Close Window \\
    \ctrl + \key{W} & Close Window (App-specific) \\
    \ctrl + \key{Q} & Quit Application \\
    \cmd + \key{H} & Hide/Minimize \\
    \textbf{Launch \& Switch} & \\
    \cmd & Activities / Launcher \\
    \cmd + \key{A} & Application Menu \\
    \alt + \key{Tab} & Switch Applications \\
\end{tabular}

\section{Tiling \& Layouts}
\textit{Container management and layout modes.}

\begin{tabular}{@{}ll@{}}

    \cmd + \key{c} & Float Active Window \\
    \cmd + \shft + \key{f} & Float All Windows (Panic) \\

% --- Section: Splitting ---
\section{Split Control}
\textit{Controls where the next window spawns.}

\begin{tabular}{@{}ll@{}}
    \cmd + \key{z} & \textbf{Split Horizontal} \\
    & \small{\textit{(Next window opens to the side)}} \\
    \cmd + \key{v} & \textbf{Split Vertical} \\
    & \small{\textit{(Next window opens below)}} \\
\end{tabular}

% --- Section: Container Manipulation ---
\section{Advanced Container Logic}
\textit{Working with Tabbed/Stacked groups.}

\begin{itemize}[leftmargin=10pt, noitemsep]
    \item \textbf{Creating:} Toggle split (\cmd+\key{z}/\key{v}), then open new window.
    \item \textbf{Converting:} Select split windows, press \cmd+\shft+\key{t} for Tabs.
    \item \textbf{Exiting:} Use \cmd+\shft+\key{Direction} to pop window out of group.
\end{itemize}

% --- Section: Workspaces ---
\section{Workspaces \& Monitors}
\textit{Multi-monitor and workspace flow.}

\begin{tabular}{@{}ll@{}}
    \textbf{Switch Workspace} & \\
    \cmd + \key{1-9} & Go to Workspace N \textit{\small(config req'd)} \\
    \cmd + \key{PgUp} / \key{PgDn} & Next / Previous Workspace \\
    \textbf{Move Window} & \\
    \cmd + \shft + \key{1-9} & To Workspace N \\
    \cmd + \shft + \key{PgUp/PgDn} & To Next / Previous \\
    \cmd + \shft + \key{Home/End} & To First / Last \\
    \textbf{Multi-Monitor} & \\
    \cmd + \shft + \key{h} \key{l} & Move to Monitor Left/Right \\
    \cmd + \key{x} & Flash Focus Border \\
    & \textit{\small(Find cursor across screens)} \\
\end{tabular}

\vspace{4pt}

% --- Section: Troubleshooting ---
\section{Troubleshooting}
\textit{When windows get stuck or behave oddly.}

\begin{tabular}{@{}ll@{}}
    \cmd + \key{c} & Force Float Window \\
    \cmd + \shft + \key{f} & Float All, re-tile with \cmd+\key{w} \\
    \alt + \key{F2}, \key{r} & Restart GNOME Shell \\
    & \textit{\small(X11 only; Wayland: logout/login)} \\
\end{tabular}

\vspace{3pt}
\small
\textbf{Note:} Split keys (\cmd+\key{z}/\key{v}) set a flag on the \textit{container}—the next window you open or move will spawn in that direction. Configure workspace number shortcuts (\cmd+\key{1-9}) in Settings → Keyboard.
